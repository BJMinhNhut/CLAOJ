#include <bits/stdc++.h>

using namespace std;

long long abs2(long long a)
{
    return a>0?a:-a;
}

int main()
{


    int n;
    long long *a,*b;
    cin>>n;
    a = new long long[n];
    b = new long long[n];

    for(int i=0; i<n; i++)
        cin>>a[i];
    for(int i=0; i<n; i++)
        cin>>b[i];


    sort(a+0,a+n);
    sort(b+0,b+n);

    long long min = abs(a[0] + b[0]);

    int dau = 0, cuoi = n-1;
    while(dau < n && cuoi >= 0)
    {
        if(abs2(a[dau] + b[cuoi]) < min)
            min = abs2(a[dau] + b[cuoi]);
        if((a[dau] + b[cuoi]) > 0)
            cuoi--;
        else dau++;
    }


    cout<<min<<endl;
    delete []a;
    delete []b;

    return 0;
}
